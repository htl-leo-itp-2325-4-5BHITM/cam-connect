        * resolution
    }

        --
        * windblocker: boolean
        * wireless: boolean
        * needsRecorder: boolean
    }

    entity LightType {

        * watts: int

        * variableTemperature: boolean

        * rentEndActual: Date
    }

    entity Tag {
        * tagId
        --
        * typeId: Int
        * description: string
    }

    entity Student {
        * studentId
        --
        * name: string
        * class: string
        * password
        * userid
    }

    entity Teacher {
        * teacherId
        --
        * name: string
        * verfication: string
        * password
        * userId
    }

    Device }o-down-|| DeviceType
    Device }o-down-|{ Set
